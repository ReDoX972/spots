def computeM(spots, T):
    N = len(spots)
    M = [[0 for t in range(T+1)] for k in range(N+1)]
 
    # BASE
    for t in range(T+1): M[0][t] = 0
 
    for k in range(1, N+1):
        for t in range(T+1):
            M[k][t] = M[k - 1][t]
            if t - spots[k-1][1] >= 0:
                if M[k - 1][t - spots[k-1][1]] + spots[k-1][2] > M[k][t]:
                    M[k][t] = M[k - 1][t - spots[k-1][1]] + spots[k-1][2]
 
    return M
